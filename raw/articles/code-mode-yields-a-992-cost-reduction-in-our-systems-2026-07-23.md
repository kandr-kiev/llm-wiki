// Returns per-item run stats + DEAD/FAILING/OK flags + a markdown report of flagged items.
export default async function (args: any, ctx: any) {
  const runLimit = args?.runLimit ?? 25;
  const staleDays = args?.staleDays ?? 14;
  const includeSchedules = args?.includeSchedules ?? true;
  const now = Date.now();
  const dayMs = 86400000;
  const isFail = (s: string) =&gt; /fail|error|halt|timeout|abort/i.test(String(s || &quot;&quot;));

  // ---- Workflows: run stats require workflow_listRuns (no lastRunAt on the row) ----
  const wl = await ctx.swarm.workflow_list({});
  const wfs: any[] = wl?.data ?? [];
  const rows: any[] = [];
  const pool = 6; // small concurrency to avoid hammering
  for (let i = 0; i &lt; wfs.length; i += pool) {
    const batch = wfs.slice(i, i + pool);
    const res = await Promise.all(batch.map(async (w: any) =&gt; {
      let runs: any[] = [];
      try { const rr = await ctx.swarm.workflow_listRuns({ workflowId: w.id, limit: runLimit }); runs = rr?.data ?? []; }
      catch { runs = []; }
      const total = runs.length;
      const failed = runs.filter((r: any) =&gt; isFail(r.status)).length;
      const stamps = runs.map((r: any) =&gt; Date.parse(r.startedAt || r.createdAt || r.finishedAt || &quot;&quot;)).filter((n: number) =&gt; !isNaN(n));
      const lastRun = stamps.length ? Math.max(...stamps) : null;
      const daysSince = lastRun ? Math.round((now - lastRun) / dayMs * 10) / 10 : null;
      const failRatio = total ? Math.round(failed / total * 100) / 100 : 0;
      const dead = !w.enabled || total === 0 || (lastRun != null &amp;&amp; daysSince! &gt; staleDays);
      const failing = total &gt;= 2 &amp;&amp; failRatio &gt;= 0.5;
      return {
        kind: &quot;workflow&quot;, id: w.id, name: w.name, enabled: w.enabled,
        runs: total, failed, failRatio,
        lastRunAt: lastRun ? new Date(lastRun).toISOString() : null,
        daysSinceLastRun: daysSince, nodeCount: w.nodeCount,
        flag: failing ? &quot;FAILING&quot; : dead ? &quot;DEAD&quot; : &quot;OK&quot;,
      };
    }));
    rows.push(...res);
  }

  // ---- Schedules: lastRunAt + consecutiveErrors already on the row ----
  const scRows: any[] = [];
  if (includeSchedules) {
    const sl = await ctx.swarm.schedule_list({});
    const scs: any[] = sl?.data?.schedules ?? [];
    for (const s of scs) {
      const lastRun = s.lastRunAt ? Date.parse(s.lastRunAt) : null;
      const daysSince = lastRun ? Math.round((now - lastRun) / dayMs * 10) / 10 : null;
      const dead = !s.enabled || (!lastRun) || (daysSince != null &amp;&amp; daysSince &gt; staleDays);
      const failing = (s.consecutiveErrors ?? 0) &gt;= 3;
      scRows.push({
        kind: &quot;schedule&quot;, id: s.id, name: s.name, enabled: s.enabled,
        consecutiveErrors: s.consecutiveErrors ?? 0,
        lastRunAt: s.lastRunAt ?? null, daysSinceLastRun: daysSince,
        nextRunAt: s.nextRunAt ?? null, cron: s.cronExpression,
        flag: failing ? &quot;FAILING&quot; : dead ? &quot;DEAD&quot; : &quot;OK&quot;,
      });
    }
  }

  const rank = (f: string) =&gt; f === &quot;FAILING&quot; ? 0 : f === &quot;DEAD&quot; ? 1 : 2;
  rows.sort((a, b) =&gt; rank(a.flag) - rank(b.flag) || (b.failRatio - a.failRatio));
  scRows.sort((a, b) =&gt; rank(a.flag) - rank(b.flag) || (b.consecutiveErrors - a.consecutiveErrors));

  const sum = (arr: any[]) =&gt; ({
    total: arr.length,
    failing: arr.filter(r =&gt; r.flag === &quot;FAILING&quot;).length,
    dead: arr.filter(r =&gt; r.flag === &quot;DEAD&quot;).length,
    ok: arr.filter(r =&gt; r.flag === &quot;OK&quot;).length,
  });

  // markdown report of only the flagged (non-OK) items
  const flaggedW = rows.filter(r =&gt; r.flag !== &quot;OK&quot;);
  const flaggedS = scRows.filter(r =&gt; r.flag !== &quot;OK&quot;);
  let report = `## Workflow triage — flagged\n`;
  report += `Workflows: ${sum(rows).failing} failing / ${sum(rows).dead} dead / ${sum(rows).ok} ok (of ${rows.length})\n`;
  // ... table-building omitted here for length; full source has per-row markdown
  //     tables for both workflows and schedules — see the script catalog entry.

  return {
    params: { runLimit, staleDays, includeSchedules },
    workflows: { summary: sum(rows), rows },
    schedules: includeSchedules ? { summary: sum(scRows), rows: scRows } : null,
    report,
  };
}</code></pre><p class="text-[15px] text-zinc-600 leading-relaxed mb-6">There&#x27;s no framework here. It&#x27;s a <code>for</code> loop with a concurrency pool of 6, a couple of <code>Date.parse</code> calls, and a sort. The only thing that makes it “code mode” rather than “26 tool calls” is <em>where it runs</em>: inside the script sandbox, not inside the agent&#x27;s context. Every<!-- --> <code>ctx.swarm.workflow_listRuns</code> call above returns its (large, sometimes limit-ignoring) payload straight back into the loop variable <code>runs</code> — never into the conversation. The agent that eventually calls this script sees only the<!-- --> <code>return</code> statement at the bottom: one object, 25,811 characters, done.</p><h2 class="text-2xl font-bold text-zinc-900 mt-12 mb-4">The mechanism, in one sentence</h2><blockquote class="border-l-2 border-amber-700 pl-5 my-6 text-[16px] italic text-zinc-600">“Intermediate results stay in the execution environment by default... the agent only sees what you explicitly log or return.”<!-- --> <span class="not-italic text-[13px] text-zinc-400">—<!-- --> <a href="https://www.anthropic.com/engineering/code-execution-with-mcp" target="_blank" rel="noopener noreferrer" class="blog-external-link"><span>Anthropic, Code execution with MCP</span><svg xmlns="http://www.w3.org/2000/svg" width="24" height="24" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2" stroke-linecap="round" stroke-linejoin="round" class="lucide lucide-external-link blog-external-link-icon" aria-hidden="true"><path d="M15 3h6v6"></path><path d="M10 14 21 3"></path><path d="M18 13v6a2 2 0 0 1-2 2H5a2 2 0 0 1-2-2V8a2 2 0 0 1 2-2h6"></path></svg></a></span></blockquote><p class="text-[15px] text-zinc-600 leading-relaxed mb-6">That&#x27;s precisely what <code>runs</code> is doing on line after line above, and precisely why the raw-path column in our table is ~126x bigger than the script column — none of it is a framework trick, it&#x27;s just that raw data stopped being handed to the model 24 times in a row.</p><h2 class="text-2xl font-bold text-zinc-900 mt-12 mb-4">What it&#x27;s worth across models</h2><p class="text-[15px] text-zinc-600 leading-relaxed mb-6">The table above prices everything at Claude Sonnet 5&#x27;s rate, since that&#x27;s the model this measurement ran on.
