class QuestGenerator:
    """Генератор квестів з шаблонів."""

    def __init__(self, templates: Optional[List[Dict]] = None):
        self.templates = templates or QUEST_TEMPLATES

    def generate(self) -> Dict:
        """Згенерувати випадковий квест."""
        template = random.choice(self.templates)
        situation = random.choice(template['situations'])

        return {
            'id': template['id'],
            'type': template['type'],
            'situation': situation,
            'choices': template['choices'],
        }

    def generate_with_purity(self) -> Tuple[Dict, int]:
        """Згенерувати квест з випадковою чистотою."""
        quest = self.generate()
        purity = random.randint(1, 10)
        return quest, purity

    def generate_batch(self, count: int = 10) -> List[Dict]:
        """Згенерувати пакет квестів."""
        return [self.generate() for _ in range(count)]
