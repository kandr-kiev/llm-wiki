"""Health advice pool — randomized health recommendations based on Kp index."""


def get_health_advice(kp: float) -> str:
    """Get health advice based on Kp index.
    
    Kp < 3: minimal advice
    Kp 3-4: light advice
    Kp 5-6: moderate advice
    Kp >= 7: strong advice
    """
    if kp < 3:
        return "🩺 **Вплив на здоров'я:**\n   🌿 Спокійний день. Ніяких особливих рекомендацій — живи як зазвичай."
    
    if kp < 5:
        return "🩺 **Вплив на здоров'я:**\n   🔹 Загальний стан: мінімальний вплив, відчуй себе добре\n   🔹 Серцеві хворі: звичайний режим\n   🔹 Лікування: без змін"
    
    if kp < 7:
        return "🥴 **Вплив на здоров'я людини:**\n 🔹 Загальний стан: можливі головний біль, тиск, безсоння\n 🔹 Серцеві хворі: контроль тиску, лікар при необхідності\n 🔹 Люди з мігренями/епілепсією: підвищений ризик\n 🔹 Вагітні: уникати перевантажень\n 🔹 Діти та літні: обмежити електроніку"
    
    return "🥴 **Вплив на здоров'я людини:**\n 🔹 Загальний стан: СИЛЬНА магнітна буря — можливі головний біль, нудота, безсоння\n 🔹 Серцеві хворі: ОБОВ'ЯЗКОВИЙ контроль тиску, зв'язатися з лікарем\n 🔹 Люди з мігренями/епілепсією: ВИСОКИЙ ризик — мінімізувати навантаження\n 🔹 Вагітні: уникати будь-яких перевантажень\n 🔹 Діти та літні: повне обмеження електроніки, спокій\n 🔹 Всі: пити більше води, відпочивати"


def get_tech_impact(kp: float) -> str:
    """Get technical impact advice based on Kp index."""
    if kp < 3:
        return "🔧 **Технічний вплив:**\n 📡 Радіозв'язок: без перешкод\n 🛰 Супутники: стабільна робота\n 🛩 БПЛА/дрони: GPS стабільний"

        return "🔧 **Технічний вплив:**\n 📡 Радіозв'язок: легкі перешкоди HF-діапазону\n 🛰 Супутники: можливі незначні збої\n 🛩 БПЛА/дрони: перевіряйте GPS"
