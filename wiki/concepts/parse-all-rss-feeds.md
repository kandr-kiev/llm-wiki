"""Debug news categorization — scoring analysis."""
from src.services.news import (
fetch_news, _score_item, _assign_categories,
WAR_KEYWORDS, GEOPOL_KEYWORDS, ECONO_KEYWORDS, TECH_KEYWORDS,
CRIME_KEYWORDS, LLM_KEYWORDS, NEGATIVE_MAP,
FEEDS_LOCAL, FEEDS_INTERNATIONAL, _parse_feed, _deduplicate_headlines
)
import json
news = fetch_news()
print("=== Категоризація ===")
for cat, items in news.items():
print(f'{cat}: {len(items)}')
for item in items:
print(f' - {item}')
# Parse all RSS feeds
all_items = []
seen_titles = set()
for feed_url, source_name in FEEDS_LOCAL + FEEDS_INTERNATIONAL:
items = _parse_feed(feed_url)
for rec in items:
if len(rec) == 3:
title, desc, pub_dt = rec
else:
title, desc = rec
if title not in seen_titles:
all_items.append((title, desc))
seen_titles.add(title)
print(f"\n=== Всі RSS новини: {len(all_items)} ===")
for title, desc in all_items:
scores = {}
for cat_key, keywords in [
('war', WAR_KEYWORDS), ('geopol', GEOPOL_KEYWORDS),
('economy', ECONO_KEYWORDS), ('tech', TECH_KEYWORDS),
('crime', CRIME_KEYWORDS), ('llm', LLM_KEYWORDS)
]:
score, matched = _score_item(keywords, title, desc)
neg_keywords = NEGATIVE_MAP.get(cat_key, [])
neg_penalty = sum(5 for neg in neg_keywords if neg.lower() in title.lower())
final_score = max(0, score - neg_penalty)
scores[cat_key] = final_score
best_cat = max(scores, key=scores.get)
best_score = scores[best_cat]
status = "CATEGORIZED" if best_score >= 3 else "GENERAL"
print(f"\nTitle: {title[:80]}...")
print(f" Scores: {dict(scores)}")
print(f" Best: {best_cat} = {best_score} -> {status}")
